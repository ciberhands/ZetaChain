## Component: Swap

![](/img/docs/components-swap.png)

```tsx filename="src/app/page.tsx"
"use client";

import { ConnectButton } from "@rainbow-me/rainbowkit";
import { Swap, useEthersSigner, useZetaChainClient } from "@fadeev/swap/dist";
import { useAccount, useChainId, useWalletClient } from "wagmi";

const contract = "0xb459F14260D1dc6484CE56EB0826be317171e91F";

const Page = () => {
  const account = useAccount();
  const chainId = useChainId();
  const { data: walletClient } = useWalletClient({ chainId });
  const signer = useEthersSigner({ walletClient });
  const client = useZetaChainClient({ network: "testnet", signer });

  return (
    <div>
      <ConnectButton />
      <div className="w-[400px]">{client && <Swap contract={contract} client={client} account={account} />}</div>
    </div>
  );
};

export default Page;
```
